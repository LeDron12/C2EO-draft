## 14. break

- C

```c
#include <stdio.h>
#include <stdlib.h>

int breakOrReturn(int a)
{
    while (1)
    {
        if ((a % 5) == 0)
        {
            break;
        }

        if ((a % 3) == 0)
        {
            return 1;
        }

        a++;
    }

    printf("We broke out of the loop\n");
    return 0;
}


int main(int argc, char** argv) {
    int a = atoi(argv[1]);
    int result = breakOrReturn(a);
    printf("breakOrReturn = %d", result);
}
```

## 20. fibonacci

```c
#include <stdio.h>

int n;
int result = 1;
int lastResult = 0;

void fibonacci() {

        return;
    } else {
        tmp = result;
        result = result + lastResult;
        lastResult = tmp;
        n = n - 1;
        fibonacci();

}

int main(int argc, char** argv) {

    printf("fibonacci[%d] = ", n);
```
